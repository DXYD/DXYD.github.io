搜素文档树

find_all

soup.find_all('title')  #根据标签去找
soup.find_all('id = title')  #根据id去找
soup.find_all(class_ = "info")  #根据class去找
soup.find_all(attrs = {'float':'left'})   #根据参数
soup.find_all("div",attrs={'float','left'})  #过滤器
